270 void Drawall::line(float bX, float bY, bool writing)
271 {
272  // aX, aY: Position du point courant
273  // aG, aD: Longueur actuelle des fils.
274 
275  // bX, bY: position du point de destination.
276  // bG, bD: Longueur des fils à la destination.
277 
278  // nbPasG, nbPasD: Nombre de pas à faire pour aller à destination.
279 
280  // delaiG, delaiD: Delai avant chaque pas.
281 
282  // 1) on calcule la longueur des fils à la destination
283  // 2) on fait la différence par rapport à la longueur actuelle
284  // 3) Calcule le délai en fonction du rapport de longeur des fils
285 
286  // stoque la position voulue
287  // avant qu'elle soit modifiée par les limites
288  mFictivePosX = bX;
289  mFictivePosY = bY;
290 
291  // contrôle des limites, n'ecrit pas si en dehors
292  if (bX < 0 || bX > mSheetWidth || bY < 0 || bY > mSheetHeight) {
293  if (bX < 0)
294  bX = 0;
295  else if (bX > mSheetWidth)
296  bX = mSheetWidth;
297 
298  if (bY < 0)
299  bY = 0;
300  else if (bY > mSheetHeight)
301  bY = mSheetHeight;
302 
303  write(false);
304  } else {
305  // s'il est dedans et qu'il doit ecrire, écrit
306  if (writing) {
307  write(true);
308  } else {
309  write(false);
310  }
311  }
312 
313  // longueur fils à la destination (en pas)
314  long bG = positionToLeftLength(bX, bY);
315  long bD = positionToRightLength(bX, bY);
316 
317  // nombre de pas à faire
318  long nbPasG = bG - mLeftLength;
319  long nbPasD = bD - mRightLength;
320 
321  bool sensGHaut = true;
322  bool sensDHaut = true;
323 
324  float delaiG;
325  float delaiD;
326 
327  long dernierTempsG;
328  long dernierTempsD;
329 
330  // calcul de la direction
331  if (nbPasG < 0) {
332  sensGHaut = false;
333  }
334 
335  if (nbPasD < 0) {
336  sensDHaut = false;
337  }
338 
339  // on a le sens
340  // donc on peut retirer le signe pour simplifier les calculs
341  nbPasG = fabs(nbPasG);
342  nbPasD = fabs(nbPasD);
343 
344  if (nbPasG > nbPasD) {
345  delaiG = mDelay;
346  delaiD = mDelay * (float(nbPasG) / float(nbPasD));
347  } else {
348  delaiD = mDelay;
349  delaiG = mDelay * (float(nbPasD) / float(nbPasG));
350  }
351 
352  dernierTempsG = micros();
353  dernierTempsD = micros();
354 
355  if (sensGHaut) {
356  digitalWrite(PIN_LEFT_MOTOR_DIR, LEFT_DIRECTION);
357  } else {
358  digitalWrite(PIN_LEFT_MOTOR_DIR, !LEFT_DIRECTION);
359  }
360 
361  if (sensDHaut) {
362  digitalWrite(PIN_RIGHT_MOTOR_DIR, RIGHT_DIRECTION);
363  } else {
364  digitalWrite(PIN_RIGHT_MOTOR_DIR, !RIGHT_DIRECTION);
365  }
366 
367  while(nbPasG > 0 || nbPasD > 0)
368  {
369  // si le delai est franchi et qu'il reste des pas à faire
370  if ((nbPasG > 0) && (micros() - dernierTempsG >= delaiG)) {
371  // stoque le temps actuel dans lastTimer
372  dernierTempsG = micros();
373 
374  // incremente ou decremente (en fonction de la direction)
375  if (sensGHaut) {
376  mLeftLength++;
377  Serial.print('L');
378  } else {
379  mLeftLength--;
380  Serial.print('l');
381  }
382 
383  // decremente le nb de pas restants
384  nbPasG--;
385 
386  // Effectue le pas
387  leftStep();
388  }
389 
390  if ((nbPasD > 0) && (micros() - dernierTempsD >= delaiD)) {
391  // stoque le temps actuel dans lastTimer
392  dernierTempsD = micros();
393 
394  // incremente ou decremente (en fonction de la direction)
395  if (sensDHaut) {
396  mRightLength++;
397  Serial.print('R');
398  } else {
399  mRightLength--;
400  Serial.print('r');
401  }
402 
403  // decremente le nb de pas restants
404  nbPasD--;
405 
406  // Effectue le pas
407  rightStep();
408  }
409  }
410 
411  mPositionX = bX;
412  mPositionY = bY;
413 }

433 {
434  mQuadraticCurveX = 0;

436  mCubicCurveX = 0;
437  mCubicCurveY = 0;

439  int longmax = 20;

441  float longX = abs(x-mPositionX);

443

445  float miniY;

448  if (longX > longmax || longY > longmax) {
449

519 void Drawall::cubicCurve(float x1, float y1, float x2, float y2, float x, float y)
520 {
521  // B(t) = P0(1-t)^3 + 3*P1*t(1-t)^2 + 3*P2*t^2*(1-t) + P3*t^3
522  // t=[0,1]
523 
524  // P0 = pt source
525  // P1-P2 = pts de controles
526  // P3 = pt cible
527 
528  float x0 = mPositionX;
529  float y0 = mPositionY;
530  float ptx, pty;
531  float t;
532 
533  for(t=0; t<=1; t+=0.01) {
534  ptx = x0*pow((1-t),3) + 3*x1*t*pow(1-t, 2) + 3*x2*pow(t, 2)*(1-t) + x*pow(t,3);
535  pty = y0*pow((1-t),3) + 3*y1*t*pow(1-t, 2) + 3*y2*pow(t, 2)*(1-t) + y*pow(t,3);
536  line(ptx, pty, true);
537  }
538 
539  // finit le dernier trajet au cas ou ça ne tombe pas juste
540  line(x, y, true);
541 
542  mQuadraticCurveX = 0;
543  mQuadraticCurveY = 0;
544  mCubicCurveX = x2;
545  mCubicCurveY = y2;
546 }

573 void Drawall::quadraticCurve(float x1, float y1, float x, float y)
574 {
575  // B(t) = P0*(1-t)^2 + 2*P1*t(1-t) + P2*t^2
576  // t=[0,1]
577 
578  // P0 = pt source
579  // P1 = pt de controle
580  // P2 = pt cible
581 
582  float x0 = mPositionX;
583  float y0 = mPositionY;
584  float ptx, pty;
585 
586  for(float t=0; t<=1; t+=0.01) {
587  ptx = x0*pow(1-t, 2) + 2*x1*t*(1-t) + x*pow(t, 2);
588  pty = y0*pow(1-t, 2) + 2*y1*t*(1-t) + y*pow(t, 2);
589  line(ptx, pty, true);
590  }
591  line(x, y, true);
592 
593  mCubicCurveX = 0;
594  mCubicCurveY = 0;
595  mQuadraticCurveX = x1;
596  mQuadraticCurveY = y1;
597 }

633 void Drawall::ellipse(float rx, float ry)
634 {
635  // ratio pour déterminer
636  // la distance du point de controle en fonction du rayon
637  float k = 0.551915;
638 
639  // pour simplifier les formules
640  float x = mPositionX;
641  float y = mPositionY;
642 
643  // se déplace à gauche du cercle
644  move(x-rx, y);
645 
646  cubicCurve(x-rx,y-ry*k , x-rx*k,y-ry , x,y-ry);
647  cubicCurve(x+rx,y-ry*k , x+rx,y);
648  cubicCurve(x+rx*k,y+ry , x,y+ry);
649  cubicCurve(x-rx,y+ry*k , x-rx,y);
650 
651  // revient au centre du cercle
652  move(x, y);
